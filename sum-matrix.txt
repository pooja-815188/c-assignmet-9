#include<stdio.h>
int main()
{
    int row,col,i,j,sum=0;
    printf("enter row and col\n");
    scanf("%d %d",&row,&col);
    int mat1[row][col];
    printf("enter a first matrix array\n");
    for(i=0;i<row;i++)
    {
        for(j=0;j<col;j++)
        {
            scanf("%d",&mat1[i][j]);
        }
    }
     printf(" first matrix elements array\n");
    for(i=0;i<row;i++)
    {
        for(j=0;j<col;j++)
        {
            printf("%d\t",mat1[i][j]);
        }
        printf("\n");
    }
    printf("sum of mat1 is\n");
    for(i=0;i<row;i++)
    {
        for(j=0;j<col;j++)
        {
            sum=sum+mat1[i][j];
        }
    }
    printf("sum=%d\n",sum);
}
