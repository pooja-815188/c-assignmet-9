int linsearch(int arr[],int size,int key);
int main()
{
    int size,i;
    printf("enter a size\n");
    scanf("%d",&size);
    int arr[size];
    printf("enter a array elements\n");
    for(i=0;i<size;i++)
    {
        scanf("%d",&arr[i]);
    }
    int key;
    printf("enter a key\n");
    scanf("%d",&key);
    int pos=linsearch(arr,size,key);
    if(pos==0)
    {
        printf("key not found\n");
    }
    else
    {
        printf("key found in pos =%d\n",pos);
    }
    return 0;
}
