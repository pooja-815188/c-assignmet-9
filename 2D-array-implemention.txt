#include<stdio.h>
#include<stdlib.h>
int main()
{
    int row,col;
    printf("enter a row and col\n");
    scanf("%d %d",&row,&col);
    int *arr=(int*)malloc(row*col*sizeof(int));
    int i,j;
    for(i=0;i<row;i++)
    {
        for(j=0;j<col;j++)
        {
            *(arr + i*col + j)=i+j;
        }
    }
    printf("matrix elemets are\n");
     for(i=0;i<row;i++)
    {
        for(j=0;j<col;j++)
        {
            printf("%d\t",*(arr + i*col + j));
        }
        printf("\n");
    }
    free(arr);
}
