#include <stdio.h>

void moveZerosToLeft(int arr[], int size) 
{
    int j = 0;
    for (int i = 0; i < size; i++) 
    {
        if (arr[i] != 0) 
        {
            arr[j] = arr[i];
            j++;
        }
    }
    for (int i = j; i < size; i++) 
    {
        arr[i] = 0;
    }
}

int main()
{
    int arr[] = {1, 2, 0, 4, 3, 0, 5, 0};
    int size = sizeof(arr) / sizeof(arr[0]);
    moveZerosToLeft(arr, size);

    printf("\nModified Array: ");
    for (int i = 0; i < size; i++) 
    {
        printf("%d ", arr[i]);
    }
}
